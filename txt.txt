CREATE DATABASE libreriaCoceco;

CREATE TABLE boletas (
	idboletas int(5) NOT NULL AUTO_INCREMENT,
	fecha SMALLINT(2) ,
	nombre VARCHAR (10),
	coche INT(5),
	gasto INT(5),
	pos INT(5),
	efectivo INT(5),
	cuota INT(5),
	PRIMARY KEY (idboletas)
);
	
CREATE TABLE gastos (
	idgastos int(5) NOT NULL AUTO_INCREMENT,
	fecha INT(5),
	coche INT(5),
	gasto INT(5),
	detalle VARCHAR (10),
	PRIMARY KEY (idgastos)
);

CREATE TABLE socios (
	idsocios INT (5) NOT NULL AUTO_INCREMENT,
	nombre VARCHAR(10),
	apellido VARCHAR (10),
	email VARCHAR(12),
	telefono INT (10),
	fdeingreso int(2),
	legajo INT(3),
	PRIMARY KEY (idsocios)
);
